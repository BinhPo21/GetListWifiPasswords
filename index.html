# @BinhPo21 # Get List Name Wifi on Windows $wifiProfiles = netsh wlan show profiles | Select-String -Pattern "All User Profile" | ForEach-Object { $_.ToString().Split(":")[1].Trim() } # Patch $scriptDirectory = "C:\!WifiPasswords" $fileName = "listWifiPasswords.txt" # Create a directory if it does not exist if (-not (Test-Path -Path $scriptDirectory)) { New-Item -Path $scriptDirectory -ItemType Directory | Out-Null } $passwordList = @() # Get List Password with List Name Wifi Write-Host "Getting list of Wifi Passwords..." foreach ($profile in $wifiProfiles) { $profileInfo = netsh wlan show profile name="$profile" key=clear $password = $profileInfo | Select-String -Pattern "Key Content" | ForEach-Object { $_.ToString().Split(":")[1].Trim() } $passwordList += "${profile}:${password}" } if ($passwordList) { $passwordList | Out-File -FilePath "$scriptDirectory\$fileName" Write-Host "Path to the list of saved WiFi Passwords: $scriptDirectory\$fileName" Invoke-Item "$scriptDirectory\$fileName" } Read-Host "Press Enter to exit" exit
