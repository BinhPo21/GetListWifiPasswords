# @BinhPo21 # Lấy danh sách các mạng Wi-Fi đã lưu trên máy tính $wifiProfiles = netsh wlan show profiles | Select-String -Pattern "All User Profile" | ForEach-Object { $_.ToString().Split(":")[1].Trim() } # Đường dẫn đến thư mục chứa file $scriptDirectory = "C:\!WifiPasswords" # Tên file đích $fileName = "listWifiPasswords.txt" # Tạo thư mục nếu nó không tồn tại if (-not (Test-Path -Path $scriptDirectory)) { New-Item -Path $scriptDirectory -ItemType Directory | Out-Null } $passwordList = @() Write-Host "Getting list Wifi Passwords..." foreach ($profile in $wifiProfiles) { $profileInfo = netsh wlan show profile name="$profile" key=clear $password = $profileInfo | Select-String -Pattern "Key Content" | ForEach-Object { $_.ToString().Split(":")[1].Trim() } $passwordList += "${profile}:${password}" } if ($passwordList) { $passwordList | Out-File -FilePath "$scriptDirectory\$fileName" Write-Host "Path to list of saved Passwords: $scriptDirectory\$fileName" Invoke-Item "$scriptDirectory\$fileName" } Read-Host "Press Enter to exit" exit
